2.在Linux下，利用 socket 技术编写程序，包含客户端和服务端，实现两者之间的信息传递。

client

#include<stdio.h> #include<unistd.h> #include<stdlib.h> #include<sys/socket.h> #include<arpa/inet.h> #include<string.h>

#define SERV_IP “127.0.0.1” #define SERV_PORT 8888

int main(void) {

int cfd,ret;
struct sockaddr_in serv_addr;
char buf[BUFSIZ];
int n;

cfd=socket(AF_INET,SOCK_STREAM,0);
if( cfd < 0){
    perror("socket error");
    exit(1);
}

memset(&serv_addr,0,sizeof(serv_addr));//memset把serv_addr指针清成空，防止默认有的时候初始化失败，默认那个值取随机值
serv_addr.sin_family=AF_INET;
serv_addr.sin_port=htons(SERV_PORT);
ret=inet_pton(AF_INET,SERV_IP,&serv_addr.sin_addr.s_addr);
if( ret< 0){
    perror("inet_pton error");
    exit(1);
}

ret=connect(cfd,(struct sockaddr *)&serv_addr,sizeof(serv_addr));
if( ret< 0){
    perror("connect error");
    exit(1);
}

printf("send msg to server: \n");
fgets(buf,sizeof(buf),stdin);//用户输入的所有东西都读到buf里

write(cfd,buf,strlen(buf)); //写到服务器端,用fgets得到的实际字符串后面会有\n\0，所以要用strlen，‘\0’作为strlen的结束符
n=read(cfd,buf,sizeof(buf));
printf("Received: %s\n",buf);
//write(STDOUT_FILENO,buf,n);

close(cfd);
return 0;

}

server

#include<unistd.h> #include<sys/socket.h> #include<stdlib.h> #include<arpa/inet.h> #include<ctype.h> #define SERV_IP “127.0.0.1” #define SERV_PORT 8888
//调试过程中可用netstat -apn | grep 8888检查端口占用情况,有占用的话用kill + 端口号 杀掉占用 //用man + 函数名查看函数参数和用法。man 7 ip查看结构体sockaddr用法

int main(void) {

int lfd,cfd;
struct sockaddr_in serv_addr,client_addr;
socklen_t client_addr_len;
char buf[BUFSIZ],client_IP[BUFSIZ];//BUFSIZ操作系统内嵌的宏，专门指定buf大小
int n,i,ret;

lfd=socket(AF_INET,SOCK_STREAM,0);
if(lfd==-1){
    perror("socket error");
    exit(1);
}

serv_addr.sin_family=AF_INET;//地址族协议，IPv4还是ipv6
serv_addr.sin_port=htons(SERV_PORT);//端口号，但是要用htons将本地字节序转化成网络字节序，大小端转换
serv_addr.sin_addr.s_addr=htonl(INADDR_ANY);//需要将本地字节序转化为网络字节序进行大小端转换，但是要求h是数字的，需要用inet_pton，但是INADDR_ANY可以绑定本地有效的任意ip，直接就是数字类型了，所以可以>用htonl

ret=bind(lfd,(struct sockaddr *)&serv_addr,sizeof(serv_addr));
if(ret==-1){
    perror("bind error");
    exit(1);
}

ret=listen(lfd,128);
if(ret==-1){
    perror("listen error");
    exit(1);
}

client_addr_len=sizeof(client_addr);

cfd=accept(lfd,(struct sockaddr *)&client_addr,&client_addr_len); if(cfd==-1){ perror(“accept error”); exit(1); }

printf("client IP:%s, client port:%d\n",inet_ntop(AF_INET,&client_addr.sin_addr.s_addr,client_IP,sizeof(client_IP)),ntohs(client_addr.sin_port));



n=read(cfd,buf,sizeof(buf));//读来自客户端的数据，读到buf里
printf("recv msg from client:%s\n",buf);
for(i=0;i<n;i++)
    buf[i]='h';
write(cfd,buf,n);//对客户端发来的数据进行简单的处理并回写到客户端


close(lfd);
close(cfd);

return 0;

}
